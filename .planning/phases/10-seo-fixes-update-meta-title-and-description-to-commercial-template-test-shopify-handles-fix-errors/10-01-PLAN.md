```typescript
export function generateBrandMetadata(
  brand: {
    title: string;
    image?: { url: string } | null;
  },
  locale: string,
  slug: string
): Metadata {
  const isUk = locale === 'uk';
  const title = isUk
    ? `${brand.title} — купити онлайн | MioMio`
    : `${brand.title} — купить онлайн | MioMio`;
  const description = isUk
    ? 'Моделі бренду в MioMio: фото, доступні розміри та актуальна наявність. Доставка по Україні ✔️'
    : 'Модели бренда в MioMio: фото, доступные размеры и актуальное наличие. Доставка по Украине ✔️';
  return generatePageMetadata(
    { title, description, image: brand.image?.url },
    locale,
    `/brand/${slug}`
  );
}
```

```typescript
export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const { slug, locale } = await params;
  const decodedSlug = decodeURIComponent(slug);

  try {
    const { collection } = await getCollection({
      handle: decodedSlug,
      first: 1,
      locale,
    });

    if (!collection.collection) {
      return { title: 'Brand Not Found' };
    }

    return generateBrandMetadata(
      {
        title: collection.collection.title,
        image: collection.collection.image ?? null,
      },
      locale,
      decodedSlug
    );
  } catch {
    return { title: 'Brand Not Found' };
  }
}
```

Note: `collection` from `getCollection` is `GetCollectionQuery` (the Shopify result), and `collection.collection` is the actual collection object with `title`, `image: { url, altText }`. The `image` field is confirmed present in the `GetCollectionWithProducts` query.
