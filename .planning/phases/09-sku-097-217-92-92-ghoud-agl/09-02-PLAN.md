  <files>src/features/order/ui/OrderStatusBadge.tsx</files>
  <action>
    Rewrite OrderStatusBadge.tsx to handle both Shopify enum values and Ukrainian/Russian external API status strings.

    Replace the `statusVariantMap` + `variant` approach with `isGreen` / `isRed` boolean flags and Tailwind inline classes via `cn()`. Import `cn` from `@shared/lib/utils`.

    New logic:
    ```typescript
    'use client';

    import { Badge } from '@shared/ui/badge';
    import { useTranslations } from 'next-intl';
    import { cn } from '@shared/lib/utils';

    type OrderStatusBadgeProps = {
      status: string;
    };

    export const OrderStatusBadge = ({ status }: OrderStatusBadgeProps) => {
      const t = useTranslations('OrderPage.status');
      const upper = status.toUpperCase();

      const isGreen = ['FULFILLED', 'ОБРОБЛЕНО'].includes(upper);
      const isRed = ['CANCELLED', 'СКАСОВАНО', 'ВІДМОВА ВІД ОТРИМАННЯ', 'ON_HOLD', 'ОТМЕНЕН'].includes(upper);

      const getStatusLabel = (s: string): string => {
        switch (s.toUpperCase()) {
          case 'FULFILLED': return t('fulfilled');
          case 'UNFULFILLED': return t('unfulfilled');
          case 'ON_HOLD': return t('onHold');
          case 'PARTIALLY_FULFILLED': return t('partiallyFulfilled');
          case 'ОБРОБЛЕНО': return s;
          case 'СКАСОВАНО': return s;
          case 'ВІДМОВА ВІД ОТРИМАННЯ': return s;
          case 'ОТМЕНЕН': return s;
          default: return s;
        }
      };

      return (
        <Badge
          className={cn('rounded', {
            'bg-green-100 text-green-800 border-green-200': isGreen,
            'bg-red-100 text-red-800 border-red-200': isRed,
          })}
          variant={isGreen || isRed ? 'outline' : 'secondary'}
        >
          {getStatusLabel(status)}
        </Badge>
      );
    };
    ```

    Note: ON_HOLD is mapped to red (destructive) — consistent with the original intent of `destructive` variant.
    Note: ОТМЕНЕН is the Russian cancellation string returned by the external API — per locked CONTEXT.md decision, "Отменен / Скасовано status update logic — ensure handled correctly". It must appear in isRed.
    Note: PARTIALLY_FULFILLED stays as secondary (no special color) — not in the green or red lists.
  </action>

    Visually confirm the badge renders via `npm run build` not failing.
  </verify>
